https://YOUR-DOMAIN.com/api/whatsapp-lead
4

🔗 Webhook مباشر — الأسرع والأكثر تحكماً
مناسب مع WATI / 360dialog / Twilio / AiSensy / Green API
أو أي منصة تدعم إرسال HTTP POST عند وصول رسالة
1
ضع هذا الكود على سيرفرك (Node.js)
// server.js — WhatsApp Webhook for BW CRM const express = require('express'); const app = express(); app.use(express.json()); const CRM_TOKEN = process.env.WA_TOKEN || 'bw_wa_secret'; const fs = require('fs'); const DB_FILE = './leads.json'; // POST /api/whatsapp-lead app.post('/api/whatsapp-lead', (req, res) => { const { token, phone, name, message, timestamp } = req.body; // تحقق من التوكن if (token !== CRM_TOKEN) return res.status(401).json({ error: 'Unauthorized' }); // بناء الليد const lead = { id: Date.now().toString(36), name: name || 'واتساب: ' + phone, phone: phone || '', source: 'whatsapp', status: 'new', note: message || '', created: timestamp ? new Date(timestamp*1000).getTime() : Date.now(), stage: 'assign', assign_t: Date.now() }; // احفظ في ملف JSON (أو قاعدة بيانات) let leads = []; if (fs.existsSync(DB_FILE)) leads = JSON.parse(fs.readFileSync(DB_FILE)); // تجنب التكرار if (!leads.find(l => l.phone === phone)) { leads.push(lead); fs.writeFileSync(DB_FILE, JSON.stringify(leads, null, 2)); console.log('✅ New WA Lead:', name, phone); } res.json({ success: true, id: lead.id }); }); app.listen(3000, () => console.log('🚀 Webhook ready on :3000'));

https://YOUR-DOMAIN.com/api/meta-webhook

3
كود الـ Webhook لـ Meta (Node.js)
// Meta Webhook Verification (GET) app.get('/api/meta-webhook', (req, res) => { const mode = req.query['hub.mode']; const token = req.query['hub.verify_token']; const challenge = req.query['hub.challenge']; if (mode === 'subscribe' && token === VERIFY_TOKEN) return res.send(challenge); res.sendStatus(403); }); // Receive Messages (POST) app.post('/api/meta-webhook', (req, res) => { const entry = req.body?.entry?.[0]; const change = entry?.changes?.[0]?.value; const msg = change?.messages?.[0]; if (!msg) return res.sendStatus(200); const phone = msg.from; const name = change.contacts?.[0]?.profile?.name || phone; const text = msg.text?.body || ''; const ts = msg.timestamp; // أرسل للـ CRM fetch('http://localhost:3000/api/whatsapp-lead', { method: 'POST', headers: {'Content-Type':'application/json'}, body: JSON.stringify({ token: CRM_TOKEN, phone, name, message: text, timestamp: ts }) }); res.sendStatus(200); });
